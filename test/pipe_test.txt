# Combinations 1: Single commands

## Single outfile
ls tmp
>tmp/infile.tmp
ls tmp
cat tmp/infile.tmp
echo holaaa >tmp/infile.tmp
cat tmp/infile.tmp

## Single infile
<tmp/no_exists => open infile: No such file or directory
<tmp/infile.tmp

## Single command
ls
echo hola


# Combinations 2: One command

## Infile Cmmd
<tmp/infile.tmp cat
<tmp/no_exists cat => open infile: No such file or directory

## Cmmd Outfile
echo hola > tmp/outfile.tmp
cat tmp/outfile.tmp
echo adeu >> tmp/outfile.tmp
cat tmp/outfile.tmp
echo adeu > tmp/outfile.tmp

## Infile Outfile
< tmp/infile.tmp > tmp/outfile.tmp
< tmp/no_exists > tmp/outfile.tmp => open infile: No such file or directory
cat tmp/outfile.tmp
<tmp/infile.tmp cat >tmp/outfile.tmp
cat tmp/outfile.tmp

## Multiple redirection
echo hola1 > tmp/infile1.tmp > tmp/infile2.tmp
cat tmp/infile1.tmp

<tmp/infile1.tmp <tmp/infile2.tmp cat
<tmp/infile2.tmp <tmp/infile1.tmp cat
