# Combinations 1: Single commands

## Single infile
<tmp/no_exists => open infile: No such file or directory
<tmp/file.tmp

## Single outfile
ls
>tmp/outfile.tmp
ls
cat tmp/outfile.tmp
>tmp/outfile.tmp

## Single command
ls
echo hola


# Combinations 2: One command

## Infile Cmmd
<tmp/file.tmp cat
<tmp/no_exists cat => open infile: No such file or directory

## Cmmd Outfile
echo hola > tmp/outfile.tmp
cat tmp/outfile.tmp
echo adeu >> tmp/outfile.tmp
cat tmp/outfile.tmp
echo adeu > tmp/outfile.tmp

## Infile Outfile
< tmp/file.tmp > tmp/outfile.tmp
< tmp/no_exists > tmp/outfile.tmp => open infile: No such file or directory
cat tmp/outfile.tmp
<tmp/file.tmp cat >tmp/outfile.tmp
cat tmp/outfile.tmp
